create or replace package body course_package_api as
procedure getallcourse
as
c_all sys_refcursor;
begin
open c_all for
select * from course_api;
dbms_sql.return_result(c_all);
end getallcourse;
PROCEDURE createinsertcourse(idofcourse in int,nameofcourse in varchar,price in int)

begin

commit;
end createinsertcourse;

begin

commit;
end deletecourse;

create or replace package body categorey_package_api
as
procedure insertcat(categoreyname in varchar,dateofcat in date)
is
begin
insert into categorey_api values(null,categoreyname,to_date(dateofcat,'YYYY-MM-DD'));
commit;
end insertcat;
procedure getallcat
as
c_all sys_refcursor;
begin
open c_all for
select * from categorey_api;
dbms_sql.return_result(c_all);
end getallcat;
procedure updatecat(idofcat in number,categoreyname in varchar,dateofcat in date)
is
begin
update categorey_api set catname=categoreyname ,catdate=dateofcat where id=idofcat;
commit;
end updatecat;
procedure deletecat(idofcat in number)
is
begin
delete categorey_api where id=idofcat;
commit;
end deletecat;

c_all sys_refcursor;
begin
open c_all for
